#!/bin/bash

# log_review.sh
# Purpose: Search logs for important events

ADMIN_NAME="Juan"
MODE="${1:-all}"
REPORT_FILE="log_review_report.txt"
ERROR_FILE="log_review_errors.log"
DATE_NOW="$(date '+%Y-%m-%d %H:%M:%S')"
CURRENT_USER="$USER"
CUSTOM_ROLE="${CLASS_ROLE:-Not set in ~/.bashrc}"

KEYWORDS=("error" "failed" "warning")

echo "Log Review Report" > "$REPORT_FILE"
echo "Generated by: $ADMIN_NAME" >> "$REPORT_FILE"
echo "Date: $DATE_NOW" >> "$REPORT_FILE"
echo "User: $CURRENT_USER" >> "$REPORT_FILE"
echo "CLASS_ROLE: $CUSTOM_ROLE" >> "$REPORT_FILE"
echo "----------------------------------------" >> "$REPORT_FILE"

if grep -q "CLASS_ROLE=" ~/.bashrc 2>>"$ERROR_FILE"; then
    echo "Persistent variable found in ~/.bashrc" >> "$REPORT_FILE"
else
    echo "Persistent variable not found in ~/.bashrc" >> "$REPORT_FILE"
fi

search_keyword() {
    local word="$1"
    local count
    count=$(journalctl -n 200 2>>"$ERROR_FILE" | grep -i "$word" | wc -l)
    echo "Keyword: $word | Matches: $count" >> "$REPORT_FILE"

    if [ "$count" -gt 0 ]; then
        echo "Recent matches:" >> "$REPORT_FILE"
        journalctl -n 200 2>>"$ERROR_FILE" | grep -i "$word" | tail -5 >> "$REPORT_FILE"
    else
        echo "No recent matches found." >> "$REPORT_FILE"
    fi
    echo "----------------------------------------" >> "$REPORT_FILE"
}

case "$MODE" in
    error)
        search_keyword "error"
        ;;
    failed)
        search_keyword "failed"
        ;;
    warning)
        search_keyword "warning"
        ;;
    all)
        for item in "${KEYWORDS[@]}"
        do
            search_keyword "$item"
        done
        ;;
    *)
        echo "Usage: $0 {error|failed|warning|all}"
        exit 1
        ;;
esac

if git rev-parse --is-inside-work-tree &>/dev/null; then
    echo "Git status summary:" >> "$REPORT_FILE"
    git status --short >> "$REPORT_FILE" 2>>"$ERROR_FILE"
else
    echo "Not running inside a Git repository." >> "$REPORT_FILE"
fi

echo "# Collaboration comment goes below this line." >> "$REPORT_FILE"
